# HTTP server - redirect to HTTPS
server {
    listen 80;
    listen [::]:80;
    server_name skillsupslovakia.org www.skillsupslovakia.org;
    
    # Let's Encrypt challenge
    location /.well-known/acme-challenge/ {
        root /var/www/skillsupslovakia/public;
    }
    
    # Redirect all other traffic to HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}

# HTTPS server
server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name skillsupslovakia.org www.skillsupslovakia.org;
    
    # Let's Encrypt SSL certificates
    ssl_certificate /etc/letsencrypt/live/skillsupslovakia.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/skillsupslovakia.org/privkey.pem;
    
    # SSL configuration
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    
    client_max_body_size 10M;
    root /var/www/skillsupslovakia/public;
    
    # Static files with caching
    location /uploads/ {
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
    
    location /css/ {
        expires 30d;
    }
    
    location /js/ {
        expires 30d;
    }
    
    location /lib/ {
        expires 30d;
    }
    
    location /img/ {
        expires 30d;
    }
    
    # Proxy to Node.js application
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }
}
